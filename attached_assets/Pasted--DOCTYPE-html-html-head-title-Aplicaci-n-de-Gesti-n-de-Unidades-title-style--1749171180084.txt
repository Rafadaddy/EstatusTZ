      background-color: red;

    }

      background-color: green;

    }

  <h1>Aplicación de Gestión de Unidades</h1>
  <label for="unit-number">Número de Unidad:</label>
  <input type="number" id="unit-number" min="1" required>
  <button onclick="addUnit()">Agregar Unidad</button>
  <table id="unit-table">
    <thead>
      <tr>
        <th>Autobús</th>
        <th>MOT</th>
        <th>TRAN</th>
        <th>ELE</th>
        <th>A/A</th>
        <th>FRE</th>
        <th>SUS</th>
        <th>DIR</th>
        <th>HOJ</th>
        <th>TEL</th>
        <th></th>
      </tr>
    </thead>
    <tbody></tbody>
  </table>

  <script>
    let units = [];

    function addUnit() {
      const unitNumber = document.getElementById('unit-number').value;
      if (!unitNumber) {
        alert('Por favor, ingresa el número de unidad.');
        return;
      }
      const newUnit = {
        Autobús: unitNumber,
        MOT: 'listo',
        TRAN: 'listo',
        ELE: 'listo',
        'A/A': 'listo',
        FRE: 'listo',
        SUS: 'listo',
        DIR: 'listo',
        HOJ: 'listo',
        TEL: 'listo'
      };
      units.push(newUnit);
      updateTable();
      document.getElementById('unit-number').value = '';
    }

    function updateTable() {
      const tableBody = document.getElementById('unit-table').getElementsByTagName('tbody')[0];
      tableBody.innerHTML = '';

      for (let i = 0; i < units.length; i++) {
        const unit = units[i];
        const row = document.createElement('tr');
        for (const key in unit) {
          const cell = document.createElement('td');
          cell.textContent = unit[key];
          cell.classList.add(typeof unit[key] === 'string' ? unit[key] : '');
          cell.addEventListener('click', () => toggleColor(key, unit, i));
          row.appendChild(cell);
        }
        const eliminateCell = document.createElement('td');
        eliminateCell.classList.add('eliminate-row');
        eliminateCell.textContent = 'Eliminar';
        eliminateCell.addEventListener('click', () => eliminateUnit(i));
        row.appendChild(eliminateCell);
        tableBody.appendChild(row);
      }
    }

    function toggleColor(key, unit, index) {
      if (key !== 'Autobús') {
        if (unit[key] === 'listo') {
          unit[key] = 'taller';
        } else {
          unit[key] = 'listo';
        }
        updateTable();
        checkEliminateRow(index);
      }
    }

    function checkEliminateRow(index) {
      const row = document.querySelectorAll('#unit-table tbody tr')[index];
      let allListo = true;
      for (const cell of row.querySelectorAll('td:not(:last-child)')) {
        if (cell.classList.contains('taller')) {
          allListo = false;
          break;
        }
      }
      row.querySelector('.eliminate-row').style.display = allListo ? 'table-cell' : 'none';
    }

    function eliminateUnit(index) {
      units.splice(index, 1);
      updateTable();
    }
  </script>
